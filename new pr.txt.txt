1. создаем имя автора и емайл
git config --global user.name 'irina'
  git config --global user.email 'irina@bk.ru'
2.git init (создает папку.git)
3.git status (отслежив.в каком статусе находится файл)
4.git add .(отслеживает файл в репозиторий)
5.git status
6.git commit -m 'all files were added to the project' (сохраним все файлы,создаем коммит)
7.git log (посмотреть историю коммитов, сохранений)
8.git branch -a (чтобы создать новую ветку, смотрим список всех веток 
и переходим на главную)
9.git checkout master (переходим на главную ветку master)
10.git checkout -b your_name (создаем новую ветку)
11.git diff (посмотреть изменения, которые были произведены над новым файлом)
12.git status (отслежив.в каком статусе находится файл)
13.git add .(отслеживает файл в репозиторий)
14.git status
15.git commit -m 'all files were added to the project' (сохраним все файлы,создаем коммит)
16.git log (посмотреть историю коммитов, сохранений)
17.git checkout master (переходим на главную ветку master)
18.git merge your_name(соединяем ветки)
19.git log (посмотреть историю коммитов, сохранений)
